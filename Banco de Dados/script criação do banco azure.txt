CREATE TABLE empresa (
	id_empresa INT IDENTITY PRIMARY KEY,
	email VARCHAR(45),
	senha VARCHAR(45),
	nome_fantasia VARCHAR(45),
	razao_social VARCHAR(45),
	cnpj CHAR(14),
	cep CHAR(8),
	uf CHAR(2),
	cidade VARCHAR(45),
	logradouro VARCHAR(45),
	ponto_referencia VARCHAR(45),
	qtd_produtos_vendidos INT,
	total_produtos_vendidos INT,
    	autenticado BIT
);
